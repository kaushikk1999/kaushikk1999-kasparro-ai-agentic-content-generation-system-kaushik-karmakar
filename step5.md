import json

    faq = json.load(open("outputs/faq.json", "r", encoding="utf-8"))
    prod = json.load(open("outputs/product_page.json", "r", encoding="utf-8"))
    comp = json.load(open("outputs/comparison_page.json", "r", encoding="utf-8"))

    # 1) 15+ categorized questions
    qb = faq.get("question_bank", [])
    cats = {q.get("category") for q in qb if q.get("category")}
    print("question_bank:", len(qb), "categories:", len(cats))

    # 2) FAQ has 5+ Q&As
    faqs = faq.get("faqs", [])
    print("faqs:", len(faqs))

    # 3) Comparison has structured Product B
    pb = comp.get("product_b", {})
    need = ["name", "key_ingredients", "benefits", "price"]
    missing = [k for k in need if k not in pb]
    print("product_b missing:", missing)

    # 4) Fictional flag
    meta = comp.get("meta", {})
    print("product_b_fictional:", meta.get("product_b_fictional"))

    # 5) Smoke-check key GlowBoost price consistency (adapted info path)
    product_a_price = None
    for item in comp.get("comparison", []):
        if "Price" in item.get("attribute", ""):
            product_a_price = item.get("product_a_value")
            break
    print("comparison product_a price:", product_a_price)

except Exception as e:
    print(f"Error: {e}")
